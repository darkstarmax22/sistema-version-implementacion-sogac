<?php

use Illuminate\Support\Facades\Hash;

{

    public $name = '';
    public $email = '';
    public $password = '';
    public $selectedRoles = [];
    public $search = '';
    public $editingUserId = null;
    public $isModalOpen = false;

    protected function rules()
    {
        return [
            'name' => 'required|min:3|max:255',
            'email' => 'required|email|unique:users,email,' . $this->editingUserId,
            'password' => $this->editingUserId ? 'nullable|min:8' : 'required|min:8',
            'selectedRoles' => 'required|array|min:1',
        ];
    }

    public function messages()
    {
        return [
            'name.required' => 'El nombre es obligatorio.',
            'name.min' => 'El nombre debe tener al menos 3 caracteres.',
            'email.required' => 'El correo electrÃ³nico es obligatorio.',
            'email.email' => 'El formato del correo es invÃ¡lido.',
            'email.unique' => 'Este correo ya estÃ¡ registrado.',
            'password.required' => 'La contraseÃ±a es obligatoria.',
            'password.min' => 'La contraseÃ±a debe tener al menos 8 caracteres.',
            'selectedRoles.required' => 'Debes seleccionar al menos un rol.',
            'selectedRoles.min' => 'Debes seleccionar al menos un rol.',
        ];
    }

    public function openModal($id = null)
    {
        $this->resetFields();
        if ($id) {
            $this->editingUserId = $id;
            $user = User::with('roles')->find($id);
            $this->name = $user->name;
            $this->email = $user->email;
            $this->selectedRoles = $user->roles->pluck('id')->map(fn($id) => (string) $id)->toArray();
        }
        $this->isModalOpen = true;
    }

    public function closeModal()
    {
        $this->isModalOpen = false;
        $this->resetFields();
    }

    public function resetFields()
    {
        $this->name = '';
        $this->email = '';
        $this->password = '';
        $this->selectedRoles = [];
        $this->editingUserId = null;
    }

    public function save()
    {
        $this->validate();

        $userData = [
            'name' => $this->name,
            'email' => $this->email,
        ];

        if ($this->password) {
            $userData['password'] = Hash::make($this->password);
        }

        $user = User::updateOrCreate(
            ['id' => $this->editingUserId],
            $userData
        );

        // Sync roles in detalle_rol
        // Assuming estado_logico 1 by default
        $rolesToSync = [];
        foreach ($this->selectedRoles as $roleId) {
            $rolesToSync[$roleId] = ['estado_logico' => 1];
        }
        $user->roles()->sync($rolesToSync);

        $this->closeModal();
        session()->flash('message', $this->editingUserId ? 'Usuario actualizado con Ã©xito.' : 'Usuario creado con Ã©xito.');
        $this->dispatch('refresh-icons');
    }

    public function delete($id)
    {
        if ($id == auth()->id()) {
            session()->flash('error', 'No puedes eliminarte a ti mismo.');
            return;
        }
        User::find($id)->delete();
        session()->flash('message', 'Usuario eliminado permanentemente.');
        $this->dispatch('refresh-icons');
    }

    public function toggleStatus($id)
    {
        if ($id == auth()->id()) {
            session()->flash('error', 'No puedes inhabilitarte a ti mismo.');
            return;
        }
        $user = User::find($id);
        $user->update(['activo' => !$user->activo]);
        
        session()->flash('message', $user->activo ? 'Usuario habilitado correctamente.' : 'Usuario inhabilitado correctamente.');
        $this->dispatch('refresh-icons');
    }

    public function with()
    {
        return [
            'users' => User::with('roles')
                        ->where(function($query) {
                            $query->where('name', 'like', '%' . $this->search . '%')
                                  ->orWhere('email', 'like', '%' . $this->search . '%');
                        })
                        ->latest()
                        ->paginate(10),
            'allRoles' => Rol::whereIn('nombre_rol', ['profesor', 'estudiante'])->get()
        ];
    }
};
